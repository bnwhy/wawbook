✅ 3. script.js (loader + flipbook + swipe)
$(document).ready(function() {

    // SIMULATION chargement images
    setTimeout(function() {
        $("#loader").fadeOut(300);
        $("#flipbook").removeClass("hidden");

        $("#flipbook").turn({
            width: 900,
            height: 600,
            autoCenter: true,
            elevation: 50,
            gradients: true,
            duration: 800
        });

    }, 1000);

    // Navigation mobile (swipe)
    let startX = 0;

    $("#flipbook").on("touchstart", function(e) {
        startX = e.originalEvent.touches[0].clientX;
    });

    $("#flipbook").on("touchend", function(e) {
        let endX = e.originalEvent.changedTouches[0].clientX;

        if (startX - endX > 50) $("#flipbook").turn("next");
        else if (endX - startX > 50) $("#flipbook").turn("previous");
    });

});
